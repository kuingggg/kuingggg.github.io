多个空格只当一个空格



解决方法一：
用<pre>和</pre>把整个预览包起来，删去<br>换行的代码。
在HTML代码中，<pre>标签用于“原样显示”，即有几个空格，就显示几个空格，遇到回车换行符，就换行。
  1. function doPreview()
  2. {
  3.         var fdsa = document.getElementById("inputText").value;
  4.         var asdf = "<pre>";
  5.         for (var i = 0; i < fdsa.length; i++)
  6.         {
  7.                 switch(fdsa.charCodeAt(i))
  8.                 {    // 空格和回车换行就不用替换了
  9.                 case 60:
  10.                         asdf += "&lt;";
  11.                         break;
  12.                 case 62:
  13.                         asdf += "&gt;";
  14.                         break;
  15.                 case 38:
  16.                         asdf += "&amp;";
  17.                         break;
  18.                 default:
  19.                         asdf += fdsa.charAt(i);
  20.                         break;
  21.                 }
  22.         }
  23.         asdf += "</pre>";
  24.         document.getElementById('output').innerHTML = asdf;
  25.         MathJax.Hub.Queue(["Typeset",MathJax.Hub,"output"]);
  26. }
复制代码
解决方法二：
用&nbsp;替换空格。
  1. case 32:  // 空格
  2.         asdf += "&nbsp;";
  3.         break;

$x,y,z>0$

\frac xy+\frac yz+\frac zx&\geqslant\frac{9\sqrt3xyz(x^2+y^2+z^2)}{\sqrt{(xy+yz+zx)^5}};\\
\frac xy+\frac yz+\frac zx&\geqslant\sqrt{\frac{3(x+y+z)(x^2y^2+y^2z^2+z^2x^2)}{xyz(xy+yz+zx)}}.
\end{align}

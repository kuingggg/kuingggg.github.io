\begin{array}{l}
&\mathop {\lim }\limits_{n \to \infty } \frac{{\sqrt[n]{{n!}}}}{n} \\
& = e^{\mathop {\lim }\limits_{n \to \infty } \frac{1}{n}\ln \frac{{n!}}{{n^n }}}  \\
& = e^{\mathop {\lim }\limits_{n \to \infty } \frac{1}{n}\sum\limits_{i = 1}^n {\ln \frac{i}{n}} }  \\
& = e^{\int_0^1 {\ln xdx} }  \\
& = \frac{1}{e} \\
\end{array}

\begin{align*}
A\to0, B=C\to90^\circ &\riff \sin A\sin B\to0, \sin B\sin C\to1, \sin C\sin A\to0\\
& \riff \sqrt[n]{1-\sin A\sin B}\to1, \sqrt[n]{1-\sin B\sin C}\to0, \sqrt[n]{1-\sin C\sin A}\to1\\
& \riff \sum\sqrt[n]{1-\sin A\sin B}\to1+0+1=2<\frac3{\sqrt[n]4}\text{ (when $n\geqslant4$)}
\end{align*}

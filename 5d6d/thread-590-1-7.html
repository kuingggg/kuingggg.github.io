\begin{align}
&\int_0^1(1+5x^2)^ndx \\
=&\int_0^1\sum_{k=0}^nC_n^k5^kx^{2k}dx \\
=&\sum_{k=0}^nC_n^k5^k\int_0^1x^{2k}dx \\
=&\sum_{k=0}^nC_n^k\frac{5^k}{2k+1}
\end{align}
